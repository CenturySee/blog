``` yaml
    type: vless

    port: 443
    uuid: 6bcedab8-3c88-49ed-d4b9-ecef117d922f
    network: tcp
    tls: true
    udp: true
    flow: xtls-rprx-vision
    servername: aws.com # REALITY servername
    reality-opts:
      public-key: c51hGuHoYTtvq3XvqhMsUT1i2-qg9yIi8aBtiQysXBk
    client-fingerprint: firefox # cannot be empty
```

``` yaml
    uuid: 6bcedab8-3c88-49ed-d4b9-ecef117d922f
    password: YoAPEE7LoJ

    servername: example.com
    alpn:
      - h3
      - h2
      - http/1.1
    skip-cert-verify: true
    udp-relay-mode: native
```
